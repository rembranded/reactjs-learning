ES6 - New features
-----------------------
1. Lexical binding with let and const 
	Block-scoped variables
	not hoisted
	
2. Arrow functions
	setInterval(()=>console.log(new Date().toLocaleString(),1000);
	let b = a.map(item=>item*2);

3. Default args
	function add(a,b=10){
    		console.log(a+b);
	}
	add(10);   // 20
	add(10,20)  //30

4. Rest and Spread
	
5. classes and objects
	class Employee{
		constructor(id,name){
			this.id=id;
			this.name=name;
		}
		display(){
			console.log(`Id = ${this.id} Name = ${this.name}`);
		}
	}
	class Manager extends Employee{
    		constructor(id,name,projectid){
        		super(id,name);
        		this.projectid=projectid;
    		}
    		display(){
            		super.display();
            		console.log("Project Id "+this.projectid);
    		}
	}
	let mgr=new Manager(2,'Akash',101);
	mgr.display();

6. Template literals
	Backticks  ` `
	variable substituition
	string across multiple lines

7. for..of loop
	for(let item of arr)
		console.log(item);

8. Array and object destructuring
	let arr=[10,20,30];
	let [a1,a2]=arr;

	let obj={key1:10, key2: 20, key3:30, key4:40};
	let {key1:c,key4:d}=obj;
		(or)
	let c=obj.key1, d=obj.key2;

	let {key1,key4}=obj;
		(or)
	let key1=obj.key1, key4=obj.key4;
